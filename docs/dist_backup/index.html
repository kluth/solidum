Quick Example

import { atom, computed, effect } from '@sldm/core';

// Create reactive state
const count = atom(0);
const doubled = computed(() => count() * 2);

// React to changes
effect(() => {
  console.log(`Count: ${count()}, Doubled: ${doubled()}`);
});

// Update state
count(5); // Console: Count: 5, Doubled: 10
